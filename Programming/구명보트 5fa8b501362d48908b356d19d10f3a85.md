    def solution(people, limit):
        answer = 0
        people.sort()
        left = 0
        right = len(people)-1
        right_boundary = []
        while left < right:
            if people[left] + people[right] > limit:
                right_boundary.append(right)
                right -= 1 
            else:
                answer += 1 
                left += 1 
                right -= 1 
        if left == right:
            answer += 1
        answer += len(right_boundary)
        return answer
