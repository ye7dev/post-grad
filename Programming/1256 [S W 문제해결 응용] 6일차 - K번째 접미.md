    def get_kth_suffix():
        suffixes = [word[i:] for i in range(len(word))]
        suffixes.sort()
        return suffixes[K-1]
    
    T = int(input())
    for t in range(1, T+1):
        K = int(input())
        word = input()
        if len(word) < K:
            ans = "none"
        else:
            ans = get_kth_suffix()
        print(f'#{t} {ans}')
