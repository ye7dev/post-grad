    def recur(idx, cum_height):
        if idx == N:
            if cum_height >= B:
                return cum_height - B
            else:
                return float('inf')
    
        hit = recur(idx+1, cum_height + heights[idx])
        skip = recur(idx+1, cum_height)

    T = int(input())

        N, B = map(int, input().split())
        heights = list(map(int, input().split()))
        ans = recur(0, 0)
