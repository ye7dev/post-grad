    ```python
    import sys
    sys.stdin = open('temp_input/input.txt', 'r')
    
    def get_similarity():
        dp = [[0] * N for _ in range(N)]
        # dp[i][j]: X[0:i+1], Y[0:j+1] -> LCS
        # base case: single char
        dp[0][0] = 1 if X[0] == Y[0] else 0
        for j in range(1, N):
            if Y[j] == X[0] or dp[0][j-1] == 1:
                dp[0][j] = 1
        for i in range(1, N):
            if X[i] == Y[0] or dp[i-1][0] == 1:
                dp[i][0] = 1
        # iterative case
        max_sim = -1
        for i in range(1, N):
            for j in range(1, N):
                dp[i][j] = max(dp[i-1][j], dp[i][j-1])
                if X[i] == Y[j]:
                    dp[i][j] = max(dp[i][j], dp[i-1][j-1] + 1)
                else:
                    dp[i][j] = max(dp[i][j], dp[i-1][j-1])
                max_sim = max(max_sim, dp[i][j])
        return max_sim / N * 100
    
    T = int(input())
    for t in range(1, T+1):
        N = int(input())
        X = input()
        Y = input()
        ans = get_similarity()
        print(f'#{t} {ans:.2f}')
    ```
