    def matching(data):
        stack = []
        parenth_dict = {']':'[', ')':'(', '}':'{', '>':'<'}
        for p in data:
            if p in parenth_dict:
                if parenth_dict[p] != stack[-1]:
                    return False
                stack.pop()
            else:
                stack.append(p)
        if stack:
            return False
        return True
    
    for t in range(1, 11):
        case_len = int(input())
        case = input()
        ans = matching(case)
        if ans:
            print(f'#{t} {1}')
        else:
            print(f'#{t} {0}')
