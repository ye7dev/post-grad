- right child만 있고 left가 없는 경우도 있으려나 ? → 없었다
- AC 코드
    
    ```python
    class Node:
        def __init__(self, key, data, left=None, right=None):
            self.key = key
            self.data = data
            self.left = left
            self.right = right
    
    def recur(node):
        if not node:
            return
        if node.left:
            recur(node.left)
        ans.append(node.data)
        if node.right:
            recur(node.right)
    
    for t in range(1, 11):
        N = int(input())
        nodes = [Node(i, None, None, None) for i in range(0, N+1)]
        for _ in range(N):
            data = input().split()
            left_key, right_key = None, None
            if len(data) == 4:
                key, char, left_key, right_key = data
            elif len(data) == 3:
                key, char, left_key = data
            else:  # len(data) == 2:
                key, char = data
    
            cur_node = nodes[int(key)]
            cur_node.data = char
            if left_key:
                cur_node.left = nodes[int(left_key)]
            if right_key:
                cur_node.right = nodes[int(right_key)]
    
        ans = []
        recur(nodes[1])
        print(f'#{t} {"".join(ans)}')
    
    ```
